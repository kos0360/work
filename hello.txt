// Example program
#include <iostream>
#include <string>
#include <ctime>
#include <vector>
#include <string>

struct Brick{
    Brick()
    : n(0), b(0) {}
    int n;
    bool b;
};



struct Field{
    Field(int r, int c)
    : rows(r), collumn(c), store(r*c) {
        srand (time(NULL));
        }
    
    int rows;
    int collumn;
    std::vector<Brick> store;
    
    void generate(){
        for(int i=0; i<rows; ++i){
            
            for(int j=0; j<collumn; ++j){
                store[i*rows + j].n=rand()%10;
                
            }
        }
    }
};

std::ostream& operator<<(std::ostream& os, const Field& f)
{
    for(int i=0; i<f.rows; ++i){
            
            for(int j=0; j<f.collumn; ++j){
                os << f.store[i*f.rows + j].n;    
            }
            os << std::endl;
    }
    return os;
}


int main()
{
    Field f(10, 50);
    f.generate();

std::cout << f;

  std::string name;

  getline (std::cin, name);
  std::cout << "Hello, " << name << "!\n";
}
