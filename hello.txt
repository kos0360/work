struct Brick{
    Brick()
    : n(0), b(0) {}
    int n;
    bool b;
};



struct Field{
    Field(int r, int c)
    : rows(r), collumn(c) { 
        store(r*c); 
    }
    
    int rows;
    int collumn;
    std::vector<Brick> store;
    
    void generate(){
        for(int i=0; i<rows; ++i)
            for(int j=i; j<collumn; ++j){
                store[i*rows + collumn].b=rand()%1;    
            }
            
    }
};
